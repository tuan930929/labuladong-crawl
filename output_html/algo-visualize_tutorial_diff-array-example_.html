class Difference {
    constructor(nums) {
        // 差分数组
        // @visualize global
        let diff = new Array(nums.length);
        diff[0] = nums[0];
        for (let i = 1; i < nums.length; i++) {
            // @visualize color *nums[i] *nums[i-1] #9542ac
            // @visualize color *diff[i] #289c2c
            // 构造差分数组
            diff[i] = nums[i] - nums[i - 1];
            // @visualize color *nums[i] *nums[i-1] *diff[i] #unset
        }
        this.diff = diff;
    }

    increment(i, j, val) {
        // @visualize color *this.diff[i] *this.diff[j+1] #7e???? i
        // 给闭区间 [i, j] 增加 val（可以是负数）
        this.diff[i] += val;
        if (j + 1 < this.diff.length) {
            this.diff[j + 1] -= val;
        }
    }

    result() {
        let res = new Array(this.diff.length);
        res[0] = this.diff[0];
        for (let i = 1; i < this.diff.length; i++) {
            // 根据差分数组构造结果数组
            res[i] = res[i - 1] + this.diff[i];
        }
        return res;
    }
}

let nums = [1, 1, 1, 1, 1, 1];
// @visualize hide
let df = new Difference(nums);
// 区间 [1, 3] 增加 2
df.increment(1, 3, 2);
// 区间 [2, 4] 增加 3
df.increment(2, 4, 3);
// 区间 [0, 2] 减 2
df.increment(0, 2, -2);
// 得到最终结果
let result = df.result();
