var subsets = function(nums) {
    let res = [];
    // 记录走过的路径
    // @visualize status(track)
    var backtrack = function(nums, start, track) {
        // @visualize color *backtrack._vizNode #green
        res.push([...track]);
        for (let i = start; i < nums.length; i++) {
            // 做选择
            // @visualize choose(nums[i])
            track.push(nums[i]);
            // 回溯
            backtrack(nums, i + 1, track);
            // 撤销选择
            // @visualize unchoose()
            track.pop();
        }
    }
    
    backtrack(nums, 0, []);
    return res;
};

let result = subsets([1,2,3]);
