// 从 s 开始 BFS 遍历图的所有节点，且记录遍历的步数
var bfs = function(graph, s) {
    let visited = new Array(graph.size()).fill(false);
    // @visualize bfs
    let q = [];
    q.push(s);
    visited[s] = true;
    // 记录从 s 开始走到当前节点的步数
    var step = 0;

    while (q.length !== 0) {
        let sz = q.length;
        for (let i = 0; i < sz; i++) {
            let cur = q.shift();
            // @visualize color *graph._v(cur) #b????a step
            console.log("visit " + cur + " at step " + step);
            let neighbors = graph.neighbors(cur);
            for (let i = 0; i < neighbors.length; i++) {
                let w = neighbors[i];
                // @visualize color *graph._e(cur,w) #b????a step
                if (!visited[w]) {
                    q.push(w);
                    visited[w] = true;
                }
            }
        }
        step++;
    }
}

let graph = Graph.createDirectedGraphFromAdjList([
    [1, 2, 3],
    [4, 5],
    [3, 5, 6],
    [1],
    [6],
    [0],
    [4],
]);

// 从节点 0 开始进行 BFS 遍历
bfs(graph, 0);
