// 中序遍历 BST，结果有序
let res = [];
let traverse = function(root) {
    if (root == null) {
        return;
    }
    traverse(root.left);
    // 中序位置
    // @visualize color *root #6ccff9
    res.push(root.val);
    traverse(root.right);
}
