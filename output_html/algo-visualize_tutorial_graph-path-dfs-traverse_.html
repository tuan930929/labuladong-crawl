var printAllPath = function(graph, src, dest) {
    // onPath 和 path 记录当前递归路径上的节点
    var onPath = new Array(graph.size()).fill(false);
    var path = [];
    var traverse = function(graph, src, dest) {
        if (onPath[src]) {
            // 防止死循环（成环）
            return;
        }
        // 前序位置
        // @visualize color *graph._v(src) #7ed67e
        onPath[src] = true;
        path.push(src);
        if (src === dest) {
            console.log("find path: " + path.join("->"));
        }
        for (var w of graph.neighbors(src)) {
            // @visualize color *graph._e(src,w) #e77427
            traverse(graph, w, dest);
            // @visualize color *graph._e(src,w) #unset
        }
        // 后序位置
        // @visualize color *graph._v(src) #unset
        onPath[src] = false;
        path.pop();
    }

    traverse(graph, src, dest);
}

let graph = Graph.createDirectedGraphFromAdjList(
    [[3, 1], [2, 4], [3], [1, 4], [0]]
);

printAllPath(graph, 0, 4);
