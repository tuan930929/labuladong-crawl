var checkInclusion = function(t, s) {
    // 把字符串转化成数组，方便可视化
    s = s.split('');
    // 判断 s 中是否存在 t 的排列
    let need = new Map(), window = new Map();
    for (let c of t) {
        need.set(c, (need.get(c) || 0) + 1);
    }

    let left = 0, right = 0;
    // @visualize bind s[left] s[right]
    let valid = 0;
    while (right < s.length) {
        let c = s[right];
        // @visualize color *s[right] #bc4fa8
        right++;
        // 进行窗口内数据的一系列更新
        if (need.has(c)) {
            window.set(c, (window.get(c) || 0) + 1);
            if (window.get(c) === need.get(c)) {
                valid++;
            }
        }

        // 判断左侧窗口是否要收缩
        while (right - left >= t.length) {
            // 在这里判断是否找到了合法的子串
            if (valid === need.size) {
                return true;
            }
            // @visualize color *s[left] #unset
            let d = s[left];
            left++;
            // 进行窗口内数据的一系列更新
            if (need.has(d)) {
                if (window.get(d) === need.get(d)) {
                    valid--;
                }
                window.set(d, window.get(d) - 1);
            }
        }
    }
    // 未找到符合条件的子串
    return false;
};
