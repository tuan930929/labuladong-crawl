var removeElement = function(nums, val) {
    let fast = 0, slow = 0;
    while (fast < nums.length) {
        if (nums[fast] !== val) {
            nums[slow] = nums[fast];
            // @visualize color *nums[slow] #62d95c
            slow++;
        }
        fast++;
    }
    return slow;
};

let result = removeElement([1,2,3,3,4,5,5], 3);
