var validTree = function(n, edges) {
    // 初始化 0...n-1 共 n 个节点
    const uf = UF.create(n);
    // 遍历所有边，将组成边的两个节点进行连接
    for (let edge of edges) {
        let u = edge[0];
        let v = edge[1];
        // 若两个节点已经在同一连通分量中，会产生环
        if (uf.connected(u, v)) {
            // @visualize color *graph._e(u,v) #c3364b
            return false;
        }
        // 这条边不会产生环，可以是树的一部分
        // @visualize color *graph._e(u,v) #167e24
        uf.union(u, v);
    }
    // 要保证最后只形成了一棵树，即只有一个连通分量
    return uf.count() === 1;
};
