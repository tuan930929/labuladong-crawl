var levelOrderTraverse = function(root) {
    if (root === null) {
        return [];
    }
    let q = [root];
    // 记录当前遍历到的层数（根节点视为第 1 层）
    let depth = 1;
    while (q.length > 0) {
        let sz = q.length;
        for (let i = 0; i < sz; i++) {
            let cur = q.shift();
            // 访问 cur 节点，同时知道它所在的层数
            // @visualize color *cur #6??ff9 depth
            console.log(`depth = ${depth}, val = ${cur.val}`);
            if (cur.left !== null) {
                q.push(cur.left);
            }
            // 把 cur 的左右子节点加入队列
            if (cur.right !== null) {
                q.push(cur.right);
            }
        }
        depth++;
    }   
};
