var isPalindrome = function(head) {

    // 先用快慢双指针找到单链表中点
    let slow = head, fast = head;
    while (fast != null && fast.next != null) {
        slow = slow.next;
        fast = fast.next.next;
    }
    if (fast != null) {
        slow = slow.next;
    }

    // 将后半部分链表反转
    let left = head;
    let right = reverse(slow);

    // 现在 left, right 同向而行就能判断回文了
    while (right != null) {
        if (left.val !== right.val) {
            return false;
        }
        left = left.next;
        right = right.next;
    }

    return true;
};

var reverse = function(head) {
    let pre = null, cur = head;
    while (cur != null) {
        // @visualize color *cur #6c8c3f
        let next = cur.next;
        cur.next = pre;
        pre = cur;
        cur = next;
    }
    return pre;
};
