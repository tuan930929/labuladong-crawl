var lengthOfLongestSubstring = function(s) {
    s = s.split('');
    let window = new Map();

    let left = 0, right = 0;
    // @visualize bind s[left] s[right]
    // 记录结果
    let res = 0;
    while (right < s.length) {
        // @visualize color *s[right] #bc4fa8
        let c = s[right];
        right++;
        // 进行窗口内数据的一系列更新
        window.set(c, (window.get(c) || 0) + 1);
        // 判断左侧窗口是否要收缩
        while (window.get(c) > 1) {
            // @visualize color *s[left] #unset
            let d = s[left];
            left++;
            // 进行窗口内数据的一系列更新
            window.set(d, window.get(d) - 1);
        }
        // 在这里更新答案
        res = Math.max(res, right - left);
    }
    return res;
};

let result = lengthOfLongestSubstring("abcabcbb");
