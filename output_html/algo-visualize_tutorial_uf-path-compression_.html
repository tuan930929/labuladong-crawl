// 示例：使用路径压缩优化并查集算法

// 创建一个大小为 6 的并查集，使用路径压缩优化
let uf = UF.createPathCompression(6)
// 显示并查集的底层数组
uf.showArray()

// 每次合并节点时，路径压缩都会将树的高度降低
// 经过路径压缩优化，树高最终将保持在常数级别

uf.union(0, 1)
uf.union(1, 2)
uf.union(2, 3)
uf.union(4, 5)
uf.union(3, 4)

// 此时，你会看到树高有退化的趋向
// 因为上面的 union 操作并未受到树高的影响，未触发路径压缩
// 下面的 connected 操作会触发路径压缩，将树高降低到常数级别

// 测试节点 0 和节点 1 的连通性，打印 true
// 时间复杂度为 O(1)
console.log('0 <-> 1:', uf.connected(0, 1))
