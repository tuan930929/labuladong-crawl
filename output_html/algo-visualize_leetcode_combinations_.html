var combine = function(n, k) {
    let res = [];
    
    // @visualize status(track)
    var backtrack = function(n, k, start, track) {
        // 到达树的底部
        if (k === track.length) {
            // @visualize color *backtrack._vizNode #green
            res.push([...track]);
            return;
        }
        // 注意 i 从 start 开始递增
        for (let i = start; i <= n; i++) {
            // 做选择
            // @visualize choose(i)
            track.push(i);
            backtrack(n, k, i + 1, track);
            // 撤销选择
            // @visualize unchoose()
            track.pop();
        }
    }
    
    if (k <= 0 || n <= 0) return res;
    let track = [];
    backtrack(n, k, 1, track);
    return res;
};

let result = combine(4, 2);
