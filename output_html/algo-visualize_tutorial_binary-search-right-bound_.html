var right_bound = function(nums, target) {
    // @visualize shape nums rect
    let left = 0, right = nums.length - 1;
    // @visualize bind nums[left]
    // @visualize bind nums[right]
    // @visualize color nums[left] nums[right] #bc4fa8
    // @visualize color nums[mid] #b47f46

    while (left <= right) {
        let mid = left + Math.floor((right - left) / 2);
        if (nums[mid] < target) {
            left = mid + 1;
        } else if (nums[mid] > target) {
            right = mid - 1;
        } else if (nums[mid] === target) {
            // 这里改成收缩左侧边界即可
            left = mid + 1;
        }
    }
    // 这里改为检查 right 越界的情况，见下图
    if (right < 0 || nums[right] !== target) {
        return -1;
    }
    return right;
};
