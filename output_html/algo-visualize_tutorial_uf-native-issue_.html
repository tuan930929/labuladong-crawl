// 示例：未优化的 UF 算法可能产生性能问题

// 创建一个未优化的并查集结构，包含 6 个节点
let uf = UF.createNaive(6)
// 显示并查集的底层数组
uf.showArray()


// 未优化的 UF 算法不会考虑合并后的树的深度
// 所以对于特定顺序的 union 操作，可能让森林退化成链表
// 这样一来，union, connected, find 操作的时间复杂度就退化成了 O(n)
// 下面就举一个例子

uf.union(0, 1)
uf.union(1, 2)
uf.union(2, 3)
uf.union(4, 5)
uf.union(3, 4)

// 测试节点 0 和节点 1 的连通性，打印 true
// 此时 connected 方法相当于遍历单链表，时间复杂度为 O(n)
console.log('0 <-> 1:', uf.connected(0, 1))

// 返回当前并查集中连通分量的数量
console.log('Connected components:', uf.count())
