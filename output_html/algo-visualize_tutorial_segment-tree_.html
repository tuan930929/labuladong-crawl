// 初始化一棵区间求和的线段树
let nums = [1, 4, 2, 3, 1, 6, 5];
let sumTree = SegmentTree.createSum(nums);

// sum(nums[1..3]) = 4 + 2 + 3 = 9
console.log(sumTree.query(1, 3));

// sum(nums[4..5]) = 1 + 6 = 7
console.log(sumTree.query(4, 5));

// 更新索引 2（第 3 个叶子节点）的值为 5
// 线段树会更新该叶子节点到根节点路径上的所有节点的值
sumTree.update(2, 5);

// sum(nums[1..3]) = 4 + 5 + 3 = 12
console.log(sumTree.query(1, 3));
