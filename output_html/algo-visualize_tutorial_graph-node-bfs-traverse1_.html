// 图结构的 BFS 遍历，从节点 s 开始进行 BFS
var bfs = function(graph, s) {
    let visited = new Array(graph.size()).fill(false);
    // @visualize bfs
    let q = [];
    q.push(s);
    visited[s] = true;

    while (q.length !== 0) {
        let cur = q.shift();
        // @visualize color *graph._v(cur) #7ed67e
        console.log("visit " + cur);
        let neighbors = graph.neighbors(cur);
        for (let i = 0; i < neighbors.length; i++) {
            let w = neighbors[i];
            // @visualize color *graph._e(cur,w) #e77427
            if (!visited[w]) {
                // @visualize color *graph._e(cur,w) #7ed67e
                q.push(w);
                visited[w] = true;
            }
        }
    }
}


let graph = Graph.createDirectedGraphFromAdjList([
