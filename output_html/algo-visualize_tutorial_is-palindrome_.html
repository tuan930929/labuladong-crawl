var isPalindrome = function(head) {
    let left = head;
    let res = true;

    // 先利用递归函数让 right 指向链表末端
    function traverse(right) {
        if (right === null) {
            return;
        }
        traverse(right.next);
        // 然后让 right 退栈，和 left 比较
        if (left.val !== right.val) {
            res = false;
        }
        // @visualize color *left #4b93ae
        // @visualize color *right #9f435d
        left = left.next;
    }

    traverse(head);
    return res;
};
