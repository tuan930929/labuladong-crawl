var searchBST = function(root, target) {
    // @visualize color *root #f4b541
    if (root === null) {
        return null;
    }
    // 去左子树搜索
    if (root.val > target) {
        return searchBST(root.left, target);
    }
    // 去右子树搜索
    if (root.val < target) {
        return searchBST(root.right, target);
    }
    // @visualize color *root #98d22d
    return root;
};
