// @visualize status(root==null?'#':root.val)
var traverse = function(root) {
    if (root === null) {
        return;
    }

    // 进入节点
    // @visualize color *root #7ed67e
    console.log("enter", root.val);
    traverse(root.left);
    traverse(root.right);
    // 离开节点
    // @visualize color *root #unset
    console.log("leave", root.val);
}
