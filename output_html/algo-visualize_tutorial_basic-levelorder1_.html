function levelOrderTraverse(root) {
    if (!root) {
        return;
    }
    let q = [root];
    while (q.length > 0) {
        let cur = q.shift();
        // 访问 cur 节点
        // @visualize color *cur #6ccff9
        console.log(cur.val);

        // 把 cur 的左右子节点加入队列
        if (cur.left) {
            q.push(cur.left);
        }
        if (cur.right) {
            q.push(cur.right);
        }
    }
}

let _allRoot = BTree.createRoot([3,9,2,1,null,5,7])
