var levelOrderTraverse = function(root) {
    if (root === null) {
        return;
    }
    let q = [];
    q.push(root);
    while (q.length !== 0) {
        let cur = q.shift();
        // 访问 cur 节点
        // @visualize color *cur #7ed67e
        console.log(cur.val);

        // 把 cur 的所有子节点加入队列
        for (let child of cur.children) {
            q.push(child);
        }
    }
}
