function State(node, depth) {
    this.node = node;
    this.depth = depth;
    this.val = node.val;
}

var levelOrderTraverse = function(root) {
    if (root === null) {
        return;
    }
    let q = [];
    // 根节点的路径权重和是 1
    q.push(new State(root, 1));

    while (q.length !== 0) {
        let cur = q.shift();
        let node = cur.node;
        let depth = cur.depth;
        // 访问 cur 节点，同时知道它的路径权重和
        // @visualize color *node #6??ff9 depth
        console.log("depth = " + depth + ", val = " + node.val);

        // 把 cur 的左右子节点加入队列
        if (cur.node.left !== null) {
            q.push(new State(node.left, depth + 1));
        }
        if (cur.node.right !== null) {
            q.push(new State(node.right, depth + 1));
        }
    }
};

let _allRoot = BTree.createRoot([3,9,2,1,null,5,7])
