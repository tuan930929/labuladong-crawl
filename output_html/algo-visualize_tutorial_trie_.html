// 创建一棵空的字典树
let trie = Trie.create();

// 向字典树中添加若干字符串元素
trie.add("that");
trie.add("the");
trie.add("their");
trie.add("apple");
trie.add("them")


// 从字典树中查找指定字符串的最短/最长前缀
// the
console.log(trie.shortestPrefixOf("theirxyz"))
// their
console.log(trie.longestPrefixOf("theirxyz"))

// 判断字典树中是否存在包含指定前缀的元素
// true
console.log(trie.hasKeyWithPrefix("app"))
// false
console.log(trie.hasKeyWithPrefix("appz"))

// 查找字典树中所有包含指定前缀的元素
// 'that', 'the', 'their', 'them'
console.log(trie.keysWithPrefix("th"))

// 从字典树中删除指定字符串元素
trie.remove("the")

// 判断字典树中是否存在指定字符串元素
// false
console.log(trie.containsKey("the"))

// 通配符 . 可以匹配任意一个字符
// 'that', 'them'
console.log(trie.keysWithPattern("th.."))
